SELECT DISTINCT
	t.tr_title,
	pe.person
FROM position p, tracks t, people pe
WHERE
	#RESTRICTIONS
	YEAR(p.po_date) = 1970
	AND	p.po_pos = 10 
	#ASSOCIATIONS
	AND	p.po_track_id = t.tr_id
	AND T.TR_ARTIST_ID = pe.p_id
ORDER BY t.tr_title ASC
;
