SELECT
	pe.person,
	t.tr_title,
	p.po_pos,
	p.po_date

	
FROM position p, tracks t, people pe

WHERE
	#RESTRICTIONS
	t.tr_year= 1969
	AND	p.po_pos = 1
	
	#ASSOCIATIONS
	AND	t.tr_artist_id = pe.p_id
	AND	t.tr_id = p.po_track_id

GROUP BY t.tr_title

ORDER BY p.po_date ASC



;
